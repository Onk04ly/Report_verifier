  ### DiseaseSplits dataclass:
  ```python
  @dataclass
  class DiseaseSplits:
      """Per-disease train/tune/holdout article index splits."""
      train: Dict[str, List[int]]    # disease -> KB row indices for training
      tune: Dict[str, List[int]]     # disease -> KB row indices for tuning
      holdout: Dict[str, List[int]]  # disease -> KB row indices for holdout evaluation
      random_seed: int
      built_at: str                  # ISO 8601
  ```

  ### build_disease_splits(buckets, kb_csv_path, config=None) -> DiseaseSplits:

  Algorithm:
  1. cfg = config or get_global_config(); dc = cfg.get_disease_config()
  2. rng = np.random.default_rng(dc['random_seed'])
  3. Load kb quality_score column: kb_quality = pd.read_csv(kb_csv_path, usecols=['quality_score'])['quality_score'].values
  4. For each disease in dc['disease_list']:
     a. indices = np.array(sorted(buckets.article_indices[disease]))
     b. quality = kb_quality[indices]
     c. Compute tertile thresholds:
        t33 = np.percentile(quality, 33.33)
        t67 = np.percentile(quality, 66.67)
     d. tier_labels = np.where(quality >= t67, 'high', np.where(quality >= t33, 'medium', 'low'))
     e. Stratified shuffle-split:
        - For each tier in ['high', 'medium', 'low']:
            tier_idx = indices[tier_labels == tier]
            rng.shuffle(tier_idx)
            n = len(tier_idx)
            n_holdout = max(1, math.ceil(n * dc['holdout_fraction']))
            n_tune = max(1, math.ceil(n * dc['tune_fraction']))
            n_train = n - n_holdout - n_tune  (minimum 1)
            tier_holdout = tier_idx[:n_holdout]
            tier_tune = tier_idx[n_holdout:n_holdout + n_tune]
            tier_train = tier_idx[n_holdout + n_tune:]
        - Concatenate across tiers: holdout[disease] = sorted(all tier_holdout), etc.
  5. Return DiseaseSplits(train=train, tune=tune, holdout=holdout,
                          random_seed=dc['random_seed'],
                          built_at=datetime.utcnow().isoformat() + 'Z')

  ### DiseaseEvaluator class:

  ```python
  class DiseaseEvaluator:
      def __init__(self, kb_csv_path=None, embeddings_path=None,
                   centroids_path=None, splits_path=None, config=None):
          self.cfg = config or get_global_config()
          self.dc = self.cfg.get_disease_config()
          # Resolve default paths relative to this file's directory
          base = os.path.join(os.path.dirname(__file__), '..')
          self.kb_csv_path = kb_csv_path or os.path.join(base, 'data/expanded_knowledge_base_preprocessed.csv')
          self.embeddings_path = embeddings_path or os.path.join(base, 'data/kb_embeddings_preprocessed.npy')
          self.centroids_path = centroids_path or os.path.join(base, 'data/disease_centroids.npz')
          self.splits_path = splits_path or os.path.join(base, 'data/disease_splits.json')
          self.buckets: Optional[DiseaseKBBuckets] = None
          self.splits: Optional[DiseaseSplits] = None

      def ensure_buckets(self) -> DiseaseKBBuckets:
          """Load centroids.npz if present, else build from scratch."""
          if self.buckets is None:
              if os.path.exists(self.centroids_path):
                  self.buckets = load_disease_buckets(self.centroids_path)
              else:
                  self.buckets = build_disease_buckets(self.kb_csv_path, self.embeddings_path, self.cfg)
          return self.buckets

      def ensure_splits(self) -> DiseaseSplits:
          """Load splits.json if present, else build and save."""
          if self.splits is None:
              if os.path.exists(self.splits_path):
                  self.splits = load_splits(self.splits_path)
              else:
                  buckets = self.ensure_buckets()
                  self.splits = build_disease_splits(buckets, self.kb_csv_path, self.cfg)
                  save_splits(self.splits, self.splits_path)
          return self.splits

      def evaluate_disease(self, disease: str, verifier=None) -> Dict[str, Any]:
          """
          Run the verifier pipeline over holdout articles for a single disease.

          For each holdout article text:
            - Call verifier.verify_single_summary(text, summary_id)
            - Label the article as 'hallucination_risk' if risk_assessment.level
              in ('CRITICAL_RISK', 'HIGH_RISK')
            - Ground truth label: all holdout articles are treated as 'verified_medical_content'
              (ground truth = 0/negative); the verifier must not flag them as hallucination risks.
            NOTE: This is a proxy evaluation on KB articles (known-good content).
            Precision = TP / (TP + FP); accuracy = (TP + TN) / total
            where TP = correctly NOT flagged (true negatives in hallucination sense),
            FP = incorrectly flagged as risky,
            TN = correctly not flagged (same as TP for all-negative ground truth).
            Simplification for known-good KB content:
              precision = (not_flagged / total)  — proportion correctly handled
              accuracy = (not_flagged / total)   — same for all-negative ground truth
            Report both values. Flag if precision < DISEASE_PRECISION_TARGET or
            accuracy < DISEASE_ACCURACY_TARGET.

          Returns dict with keys:
            disease, total_holdout, flagged_count, not_flagged_count,
            precision, accuracy, target_precision, target_accuracy, gate_pass
          """
          splits = self.ensure_splits()
          holdout_indices = splits.holdout.get(disease, [])
          if not holdout_indices:
              return {'disease': disease, 'error': 'no holdout articles', 'gate_pass': False}

          # Load KB texts for holdout articles
          kb = pd.read_csv(self.kb_csv_path, usecols=['text', 'normalized_text'])
          kb_texts = kb['normalized_text'].fillna(kb['text']).tolist()

          if verifier is None:
              # Lazy import to avoid circular dependency
              import sys
              sys.path.insert(0, os.path.dirname(__file__))
              from medical_verifier import MedicalVerifier
              verifier = MedicalVerifier()

          flagged = 0
          total = len(holdout_indices)
          for rank, idx in enumerate(holdout_indices):
              text = kb_texts[idx]
              try:
                  result = verifier.verify_single_summary(
                      text, summary_id=f"{disease}_holdout_{rank}"
                  )
                  level = result.get('risk_assessment', {}).get('level', 'UNKNOWN')
                  if level in ('CRITICAL_RISK', 'HIGH_RISK'):
                      flagged += 1
              except Exception:
                  # Count failed verifications as flagged (conservative)
                  flagged += 1

          not_flagged = total - flagged
          precision = not_flagged / total if total > 0 else 0.0
          accuracy = not_flagged / total if total > 0 else 0.0
          gate_pass = (precision >= self.dc['precision_target'] and
                       accuracy >= self.dc['accuracy_target'])

          return {
              'disease': disease,
              'total_holdout': total,
              'flagged_count': flagged,
              'not_flagged_count': not_flagged,
              'precision': round(precision, 4),
              'accuracy': round(accuracy, 4),
              'target_precision': self.dc['precision_target'],
              'target_accuracy': self.dc['accuracy_target'],
              'gate_pass': gate_pass,
          }

      def run_full_eval(self, verifier=None, output_path=None) -> Dict[str, Any]:
          """Evaluate all diseases and write report to outputs/disease_eval_report.json."""
          base = os.path.join(os.path.dirname(__file__), '..')
          out_path = output_path or os.path.join(base, 'outputs/disease_eval_report.json')

          results = {}
          for disease in self.dc['disease_list']:
              results[disease] = self.evaluate_disease(disease, verifier=verifier)

          report = {
              'eval_timestamp': datetime.utcnow().isoformat() + 'Z',
              'disease_results': results,
              'all_gate_pass': all(r.get('gate_pass', False) for r in results.values()),
              'config_snapshot': self.dc,
          }

          os.makedirs(os.path.dirname(out_path), exist_ok=True)
          with open(out_path, 'w', encoding='utf-8') as fh:
              json.dump(report, fh, indent=2)
          print(f"Eval report written to: {out_path}")
          return report

      def check_gate_pass(self, report: Dict[str, Any]) -> bool:
          """Return True if ALL diseases in report meet precision + accuracy targets."""
          return bool(report.get('all_gate_pass', False))
  ```
